// Add these rules to your Firestore Security Rules in the Firebase Console
// Go to: Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if a user is a manager
    function isManager() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'manager';
    }

    // Helper function to check if two users are in the same department
    function isSameDepartment(userId1, userId2) {
      let user1_department = get(/databases/$(database)/documents/users/$(userId1)).data.department;
      let user2_department = get(/databases/$(database)/documents/users/$(userId2)).data.department;
      return user1_department == user2_department;
    }

    // Helper function to check if a user is signed in
    function isSignedIn() {
      return request.auth != null;
    }

    // Helper function to check if a user is an admin
    function isAdmin() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Helper function to check if a user can see an employee
    function canManagerSeeEmployee(userId) {
      return userRole(userId) == 'employee' && sameDepartment(userId);
    }

    // Helper function to check the role of a user
    function userRole(userId) {
      return get(/databases/$(database)/documents/users/$(userId)).data.role;
    }

    // Helper function to check if two users are in the same department
    function sameDepartment(userId) {
      return isSameDepartment(request.auth.uid, userId);
    }

    // Rules for users collection
    match /users/{userId} {
      // Users can read/write their own data
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Managers can read data of users in the same department
      allow list, get: if request.auth != null && isManager() && isSameDepartment(request.auth.uid, userId);
    }

    // Rules for goals, activities, alerts, and nudges
    match /goals/{goalId} {
      // Users can manage their own goals
      allow read, write: if isSignedIn() && (
        request.auth.uid == resource.data.userId ||
        (isManager() && canManagerSeeEmployee(resource.data.userId)) ||
        isAdmin()
      );
      
      allow create: if isSignedIn() && (
        request.auth.uid == request.resource.data.userId ||
        (isManager() && canManagerSeeEmployee(request.resource.data.userId)) ||
        isAdmin()
      );

      // Allow managers to perform batch queries for their team
      allow list: if isSignedIn() && isManager();
    }

    match /activities/{activityId} {
      // Users can manage their own activities
      allow read, write: if request.auth != null && request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId;
      // Managers can read activities of users in the same department
      allow list, get: if request.auth != null && isManager() && isSameDepartment(request.auth.uid, resource.data.userId);
    }

    match /alerts/{alertId} {
      // Users can manage their own alerts
      allow read, update: if isSignedIn() && (
        request.auth.uid == resource.data.userId ||
        (isManager() && canManagerSeeEmployee(resource.data.userId)) ||
        isAdmin()
      );
      
      allow create: if isSignedIn();

      // Allow managers to perform batch queries for their team
      allow list: if isSignedIn() && isManager();
    }

    match /nudges/{nudgeId} {
      // Nudges can be read by the involved employee or manager
      allow read: if request.auth != null && (request.auth.uid == resource.data.employeeId || request.auth.uid == resource.data.managerId);
      // Nudges can only be created by managers
      allow create: if request.auth != null && isManager();
    }

      
      // System can create alerts for users (this allows the app to create alerts)
      allow create: if request.auth != null && 
                   request.auth.uid == request.resource.data.userId;
      
      // Allow managers to create alerts for their team members (optional - for future manager features)
      allow create: if request.auth != null && 
                   request.resource.data.keys().hasAll(['userId', 'type', 'title', 'message']);
    }
    
    // Allow other existing collections to continue working
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
